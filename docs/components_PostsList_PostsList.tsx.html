import {
  PostsListContainer,
  PostsWrapper,
  Post,
  PostContent,
  PostTitle,
  PostActions,
  PostBody,
  Button,
} from "../../styles";

import { PostType } from "../../../types";

type PostsListProps = {
  posts: PostType[];
  onDelete: (id: number) => void;
};

/**
 * The `PostsList` component renders a list of posts.
 * 
 * @component
 * @param {Object} props - The component props.
 * @param {PostType[]} props.posts - The list of posts to display.
 * @param {Function} props.onDelete - The function to call when a post is deleted.
 * 
 * @example
 * const posts = [
 *   { id: 1, title: "Post 1", body: "Body 1", userId: 1 },
 *   { id: 2, title: "Post 2", body: "Body 2" , userId: 2},
 * ];
 * return (
 *   <PostsList posts={posts} onDelete={handleDelete} />
 * )
 * 
 * @returns {JSX.Element} The rendered component.
 */

export const PostsList = ({ posts, onDelete }: PostsListProps) => (
  <PostsListContainer>
    <PostsWrapper>
      {posts.map((post) => (
        <Post key={post.id}>
          <PostContent>
            <PostTitle>{post.title}</PostTitle>
            <PostBody>{post.body}</PostBody>
          </PostContent>
          <PostActions>
            <Button
              onClick={() => onDelete(post.id)}
              aria-label={`Delete post titled ${post.title}`}
            >
              Delete
            </Button>
          </PostActions>
        </Post>
      ))}
    </PostsWrapper>
  </PostsListContainer>
);
